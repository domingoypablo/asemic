    <script src="https://cdnjs.cloudflare.com/ajax/libs/p5.js/1.4.0/p5.js"></script>

    <script>

        function setup() {

            createCanvas(windowWidth, windowHeight);

            background(255);

        }


        function draw() {

            // This space intentionally left blank for static behavior.

        }


        function mouseMoved() {

            let numShapes = int(random(3, 8));

            let radius = random(20, 80);

            let x = mouseX;

            let y = mouseY;


            for (let i = 0; i < numShapes; i++) {

                let angle = TWO_PI / numShapes * i;

                let sx = x + cos(angle) * radius;

                let sy = y + sin(angle) * radius;

                let sw = random(1, 5);

                strokeWeight(sw);

                stroke(random(0, 100), random(0, 100), random(0, 100), 150);

                line(x, y, sx, sy);

            }

        }


        function windowResized() {

            resizeCanvas(windowWidth, windowHeight);

            background(255);

        }

    </script>
